

import javax.swing.*;
import java.awt.*;

public class FacebookStaticPage {
    public static void main(String[] args) {
        JFrame frame = new JFrame("Facebook Static Page");
        frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        frame.setSize(800, 600);
        
        JPanel panel = new JPanel();
        panel.setLayout(new BorderLayout());
        
        JLabel header = new JLabel("Welcome to Facebook", SwingConstants.CENTER);
        header.setFont(new Font("Arial", Font.BOLD, 24));
        panel.add(header, BorderLayout.NORTH);
        
        JTextArea content = new JTextArea("This is a static page for Facebook. \n\n"
                + "Here you can find various features like:\n"
                + "- News Feed\n"
                + "- Messages\n"
                + "- Notifications\n"
                + "- Profile\n");
        content.setEditable(false);
        panel.add(content, BorderLayout.CENTER);
        
        JButton likeButton = new JButton("Like");
        panel.add(likeButton, BorderLayout.SOUTH);
        
        frame.add(panel);
        frame.setVisible(true);
    }
}
